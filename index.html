Linked List Nodes & Pointers

Each node stores a value and a pointer to the next node. No contiguous memory required.

length0
Node
class Node {
  constructor(value) {
    this.value = value;
    this.next = null;
  }
}
insertAt(i, value)
function insertAt(list, i, value) {
  const node = new Node(value);
  if (i === 0) { node.next = list.head; list.head = node; return; }
  let prev = list.head;
  for (let k = 0; k < i - 1; k++) prev = prev.next;
  node.next = prev.next;
  prev.next = node;
}

Binary Search Tree Ordered Tree Structure

Each node has at most two children. Left subtree < node < right subtree, always.

function insert(node, value) {
  if (!node) return { value, left: null, right: null };
  if (value < node.value) node.left = insert(node.left, value);
  else if (value > node.value) node.right = insert(node.right, value);
  return node;
}
search(value)
function search(node, value) {
  if (!node) return false;
  if (node.value === value) return true;
  return value < node.value
    ? search(node.left, value)
    : search(node.right, value);
}
